function [SSRatioindex] = SSRatio(X, c, idx, sumd);

% Log SS Ratio Index
% k = no. of clusters
	k = length(c);
% Mean of X, i.e., global center of all data
	M = mean(X);
% BGSS, weighted sum of the squared distances between each cluster center and the global center
	BGSS = length(find(idx(:))==1:k)*(pdist([c(1:k) M]))^2;
% Sum of the intra-cluster variances
	WGSS = sum(sumd);
% Log SS Ratio index
	SSRatioindex = log(BGSS/WGSS);
end;
