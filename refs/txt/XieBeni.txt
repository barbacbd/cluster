function [XieBeni_index] = XieBeni(X, IDX, sumd);

% Xie, X.L., & Beni, G. (1991). A validity measure for fuzzy clustering.
% IEEE Transactions on Pattern Analysis and Machine Intelligence,
% 13(4), 841-846.

% numerator = mean of the squared distances of all the points with respect
% to the center of the cluster they belong to.
numerator = sum(sumd)/length(X);

% denominator = the minimum of the minimum squared distances between
% the points in the clusters
del_ij = [];
for i = 1:k;
	[rows1,~] = find(IDX==i);
	clust1 = X(rows1);
	for j = i+1:k;
		[rows2,~] = find(IDX==j);
		clust2 = X(rows2);
		min_clust_diff = min(pdist2(clust1,clust2).^2);
		del_ij = [del_ij min_clust_diff ];
	end;
end;
denominator = min(del_ij);

% Xie-Beni Index
XieBeni_index = numerator/denominator;

end;
