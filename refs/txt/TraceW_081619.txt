function TraceW_index = TraceW(Wq) 
%	===================================================== 
%	Reference: 
%	--------------------------------------------------------------------------------
%	Edwards, A.W.F., and Cavalli-Sforza, L. (1965). A method for cluster 
%	analysis. Biometrika, 56, 362-375. 
%	====================================================== 
%	Inputs: 
%	--------------------------------------------------------------------------------
%	Wq		: Within-group dispersion matrix for data
%			  clustered into q clusters
%	======================================================
	Output:
%	---------------------------------------------------------------------------------
	TraceW_index = trace(Wq);
%	======================================================
end
